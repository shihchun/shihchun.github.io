Express.js з°Ўе–®ж•ҙзҗҶ

  2019-04-23   node.js   js express.js  и©•и«– еӯ—ж•ёзөұиЁҲ: 1752еӯ— й–ұи®ҖжҷӮй–“: 4еҲҶ

Express.js

WebжҮүз”ЁжЎҶжһ¶пјҲWeb application frameworkпјүпјҢе’ҢдёҖиҲ¬HTTP serverеҠҹиғҪдёҠзҡ„е·®еҲҘе°ұжҳҜпјҢд»–еҖ‘жңүеӨҡз”ЁHTTPз¶Іи·Ҝз·ЁзЁӢпјҢжҸҗдҫӣдәҶеҸҜйқ еәҰй«ҳзҡ„ж–№жЎҲпјҢи®“й–ӢзҷјиҖ…дёҚз”ЁйҖІиЎҢзӣёе°Қеә•еұӨзҡ„з·ЁзЁӢпјҢдҪҶеҚ»д№ҹжҸҗдҫӣеҸҜд»ҘеҒҡз¶Іи·Ҝз·ЁзЁӢзҡ„еҮҪејҸеә«гҖӮ

зӣёе°Қеә•еұӨзҡ„з·ЁзЁӢеҰӮпјҡе…ұйҖҡй–ҳйҒ“д»ӢйқўпјҲCGIпјҢCommon Gateway InterfaceпјүгҖҒHTTPи«ӢжұӮи§ЈжһҗгҖҒURL routingи·Ҝз”ұи§ЈжһҗгҖӮ

дҪҶжҳҜеҰӮжһңиҰҒеҒҡеҸҰеӨ–е…¶д»–Serverзҡ„е®үе…Ёж©ҹеҲ¶пјҢеҰӮSSLпјҢеҸҜиғҪе°ұиҰҒеҸҰеӨ–жһ¶иЁӯServerз„¶еҫҢжҠҠпјҢExpressжЎҶжһ¶зҷјдҪҲеңЁServerзҡ„listen portдёҠйқўпјҲеҰӮпјҡhttpгҖҒhttpsеә«пјүгҖӮ

ExpressJSеҹәж–јNodeJSдёӯй–Ӣзҷјзҡ„WebжЎҶжһ¶пјҢдё»иҰҒзҡ„иӘҝз”Ёж–№жі•е°ҚжҮүHTTPж–№жі•еҗҚеӯ—пјҢе®№жҳ“зҗҶи§ЈгҖӮйЎһдјјзҡ„е·Ҙе…·еҰӮпјҡPHPгҖҒspring bootпјҲjavaй–ӢзҷјпјүгҖҒtomecatпјҲjavaй–ӢзҷјпјүгҖҒLaravelпјҲjavaй–ӢзҷјпјүгҖҒDjangoпјҲpythonй–ӢзҷјпјүзӯүзӯүгҖӮ

еҹәжң¬е®үиЈқиҲҮиЁӯзҪ®

жү“й–ӢcmdпјҢзҖҸиҰҪеҷЁиЁӘе•ҸChocolateyпјҲжҳҜдёҖеҖӢWindowsдёӢзҡ„и»ҹй«”еҢ…з®ЎзҗҶеҷЁпјүпјҢе°Үе®үиЈқзҡ„жҢҮд»Өеҹ·иЎҢдёҖж¬ЎгҖӮ

@"%SystemRoot%\System32\WindowsPowerShell\v1.0\powershell.exe" -NoProfile -InputFormat None -ExecutionPolicy Bypass -Command "iex ((New-Object System.Net.WebClient).DownloadString('https://chocolatey.org/install.ps1'))" && SET "PATH=%PATH%;%ALLUSERSPROFILE%\chocolatey\bin"

йҖҷжҷӮеҖҷchocolateyзҡ„binе·Із¶“еҠ е…Ҙзі»зөұз’°еўғи®Ҡж•ёиЈЎйқўпјҢжүҖд»ҘеҸҜд»ҘзӣҙжҺҘдёӢжҢҮд»Өеҝ«йҖҹе®үиЈқи»ҹй«”еҢ…

choco install nvm # NodeJSзүҲжң¬з®ЎзҗҶеҷЁ
nvm install 10.15.0 #е®үиЈқ10.15.0зҡ„зүҲжң¬
nvm on #е•ҹз”Ё

е®ҳж–№е®ҡзҫ©зҡ„йҖҷе…©еҖӢArrayпјҢз¶“йҒҺжӣҙж”№пјҢжңғеҹ·иЎҢзӣёжҮүзҡ„еҠҹиғҪеңЁHTTPиЁӯе®ҡзҡ„app.listen()зҡ„зӣЈиҒҪз«ҜеҸЈдёҠйқўгҖӮ

function UDF(x){return x+1;}
// GET method route з•¶жңүдәәиЁӘе•Ҹ '/'пјҢзҡ„жҷӮеҖҷ
app.get('/:id', function (req, res) {
    UDF(3); //дҪҝз”ЁиҮӘе®ҡзҫ©еҮҪејҸ
    // еӮійҖҒ HTTP CODE
    // зӯүеҗҢ res.status(403).send('Forbidden')
    res.sendStatus(403);
    // HTTPи«ӢжұӮ
    console.log(JSON.stringify(req.headers));
    // GETи«ӢжұӮжҷӮеҖҷ еҸ–еҫ— URL/asdf зҡ„и©ұеҫ—еҲ°asdf
    console.log(req.params.id);
    // GETи«ӢжұӮжҷӮеҖҷ еҸ–еҫ— URL?xxxxдёӯзҡ„xxxx еҒҡжҗңзҙўеҠҹиғҪеҸҜд»Ҙз”Ё
    // /search?q=graphic+NVDIA+AMD => "graphic NVDIA AMD"
    console.log(req.query.xxxxx)
    res.send('GET request to the homepage');
    // еә•еұӨжҳҜres.send, еҠҹиғҪдёҖжЁЈпјҢж”№и®ҠдәҶcontent-type
    res.json('GET request to the homepage');
});

// POST method route
app.post('/',function (req, res){
    // POSTеҸҜд»ҘдҪҝз”Ёbody-parserи§Јжһҗиҝ”еӣһзҡ„HTMLиіҮж–ҷ
    // <input name="a">3</> => req.body.a = 3
    // <input name="a[asd]">3</> => req.body.a[asd] = 3
    console.log(JSON.stringify(req.body));
});
